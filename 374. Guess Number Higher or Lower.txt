def guess(num: int) -> int:
    if num < pick:
        return 1
    elif num > pick:
        return -1
    else:
        return 0

class Solution:
    def guessNumber(self, n: int) -> int:
        low, high = 1, n

        while low <= high:
            mid = (low + high) // 2
            res = guess(mid)

            if res == 0:
                return mid
            elif res == -1:
                high = mid - 1
            else:
                low = mid + 1
